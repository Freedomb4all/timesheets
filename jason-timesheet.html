// ── TimeTrack Apps Script — Cross-Device Sync Version ───────────────────────── // Handles: timesheet records, clock state, payroll periods // New in this version: doGet reads data back so any device can sync function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Clock State ────────────────────────────────────────────────────────── if (data.action === 'clockIn') { var cs = ss.getSheetByName('Clock State') || ss.insertSheet('Clock State'); if (cs.getLastRow() === 0) { cs.appendRow(['Employee','ClockIn','ClockDate','Note','Timestamp']); cs.getRange(1,1,1,5).setFontWeight('bold').setBackground('#203864').setFontColor('#ffffff'); } // Remove any existing clock-in for this employee clearClockState(cs); cs.appendRow([data.employee, data.timeIn, data.date, data.note || '', new Date().toISOString()]); return buildResponse({ok: true, action: 'clockIn'}); } if (data.action === 'clockOut') { var cs2 = ss.getSheetByName('Clock State') || ss.insertSheet('Clock State'); clearClockState(cs2); return buildResponse({ok: true, action: 'clockOut'}); } // ── Payroll ────────────────────────────────────────────────────────────── if (data.action === 'payroll') { var ps = ss.getSheetByName('Payroll Periods') || ss.insertSheet('Payroll Periods'); if (ps.getLastRow() === 0) { setupPayrollSheet(ps); } // Check if a row for this pay period already exists — update it if so var lastRow = ps.getLastRow(); var existingRow = -1; if (lastRow >= 4) { var periodCol = ps.getRange(4, 2, lastRow - 3, 1).getValues(); for (var r = 0; r < periodCol.length; r++) { if (String(periodCol[r][0]).trim() === String(data.payPeriod).trim()) { existingRow = r + 4; // 1-indexed, offset by 3 header rows + 1 break; } } } var rowValues = [ data.employee, data.payPeriod, data.week1, data.week2, data.twoWeekTotal, data.autoCalc, data.withoutOT, data.ot, '','', data.vacation, data.reasonVaca, data.holiday, data.reasonHoliday, data.sick, data.reasonSick, '','','','' ]; if (existingRow > 0) { // Update existing row — preserve manual entry columns (I,J = 9,10 and Q-T = 17-20) var existing = ps.getRange(existingRow, 1, 1, 20).getValues()[0]; rowValues[8] = existing[8]; // Mileage Units (manual) rowValues[9] = existing[9]; // Mileage $ (manual) rowValues[16] = existing[16]; // Notes (manual) rowValues[17] = existing[17]; // Bonus (manual) rowValues[18] = existing[18]; // Reimbursement (manual) rowValues[19] = existing[19]; // NOTES (manual) ps.getRange(existingRow, 1, 1, 20).setValues([rowValues]); formatDataRow(ps, existingRow); } else { // Insert new row after header rows (row 3 = first data row) var insertAt = lastRow < 4 ? 3 : lastRow; ps.insertRowAfter(insertAt); ps.getRange(insertAt + 1, 1, 1, 20).setValues([rowValues]); formatDataRow(ps, insertAt + 1); } updateTotalRow(ps); return buildResponse({ok: true, action: 'payroll'}); } // ── Timesheet Records ──────────────────────────────────────────────────── var sheet = ss.getSheetByName('Timesheet') || ss.insertSheet('Timesheet'); if (sheet.getLastRow() === 0) { sheet.appendRow(['Date','Employee','Type','Time In','Time Out','Hours','Origin','Notes','Edit History']); sheet.getRange(1,1,1,9).setFontWeight('bold').setBackground('#203864').setFontColor('#ffffff'); } if (data.action === 'delete' && data.originalDate) { var vals = sheet.getDataRange().getValues(); for (var i = 1; i < vals.length; i++) { if (String(vals[i][0]) === data.originalDate && String(vals[i][3]) === data.originalTimeIn) { sheet.deleteRow(i + 1); return buildResponse({ok: true, action: 'deleted'}); } } return buildResponse({ok: false, error: 'Row not found for delete'}); } if (data.action === 'edit' && data.originalDate) { var vals2 = sheet.getDataRange().getValues(); for (var j = 1; j < vals2.length; j++) { if (String(vals2[j][0]) === data.originalDate && String(vals2[j][3]) === data.originalTimeIn) { sheet.getRange(j+1,1,1,9).setValues([[ data.date, data.employee, data.type, data.timeIn, data.timeOut, data.hours, data.origin, data.notes, data.editHistory ]]); return buildResponse({ok: true, action: 'edited'}); } } } // Default: append new record sheet.appendRow([ data.date, data.employee, data.type, data.timeIn, data.timeOut, data.hours, data.origin, data.notes, data.editHistory || '' ]); return buildResponse({ok: true, action: 'added'}); } catch(err) { return buildResponse({ok: false, error: err.toString()}); } } // ── doGet: Returns all data so any device can sync ─────────────────────────── function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); // Read timesheet records var sheet = ss.getSheetByName('Timesheet'); var records = []; if (sheet && sheet.getLastRow() > 1) { var data = sheet.getRange(2, 1, sheet.getLastRow()-1, 9).getValues(); data.forEach(function(row) { if (!row[0]) return; // skip empty rows records.push({ date: fmtSheetDate(row[0]), employee: String(row[1]), type: String(row[2]), timeIn: fmtSheetTime(row[3]), timeOut: fmtSheetTime(row[4]), hours: parseFloat(row[5]) || 0, origin: String(row[6]), notes: String(row[7]), editHistory: parseEditHistory(String(row[8])) }); }); } // Read clock state var cs = ss.getSheetByName('Clock State'); var clockState = null; if (cs && cs.getLastRow() > 1) { var csData = cs.getRange(2, 1, cs.getLastRow()-1, 5).getValues(); for (var i = csData.length - 1; i >= 0; i--) { if (csData[i][0]) { clockState = { in: fmtSheetTime(csData[i][1]), date: fmtSheetDate(csData[i][2]), note: String(csData[i][3]) }; break; } } } return buildCorsResponse({ok: true, records: records, clockState: clockState}); } catch(err) { return buildCorsResponse({ok: false, error: err.toString()}); } } // ── Helpers ─────────────────────────────────────────────────────────────────── function clearClockState(cs) { if (cs.getLastRow() <= 1) return; var rows = cs.getRange(2, 1, cs.getLastRow()-1, 1).getValues(); for (var i = rows.length - 1; i >= 0; i--) { if (rows[i][0]) cs.deleteRow(i + 2); } } function parseEditHistory(str) { if (!str) return []; return str.split(' | ').filter(Boolean).map(function(s) { var idx = s.indexOf(': '); if (idx === -1) return {at: '', summary: s}; return {at: s.slice(0, idx), summary: s.slice(idx + 2)}; }); } // Sheets stores dates as Date objects — convert to YYYY-MM-DD string function fmtSheetDate(val) { if (!val) return ''; if (val instanceof Date) { var y = val.getFullYear(); var m = String(val.getMonth() + 1).padStart(2, '0'); var d = String(val.getDate()).padStart(2, '0'); return y + '-' + m + '-' + d; } // Already a string — strip anything past 10 chars (YYYY-MM-DD) var s = String(val).trim(); // Handle MM/DD/YYYY format that Sheets sometimes returns as text var mdyMatch = s.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/); if (mdyMatch) { return mdyMatch[3] + '-' + mdyMatch[1].padStart(2,'0') + '-' + mdyMatch[2].padStart(2,'0'); } return s.slice(0, 10); } // Sheets stores times as Date objects (fraction of a day) — convert to HH:MM string function fmtSheetTime(val) { if (!val) return ''; if (val instanceof Date) { var h = String(val.getHours()).padStart(2, '0'); var m = String(val.getMinutes()).padStart(2, '0'); return h + ':' + m; } var s = String(val).trim(); if (!s || s === '0' || s === 'null') return ''; // Already HH:MM format if (/^\d{2}:\d{2}$/.test(s)) return s; // Try parsing as a number (Sheets time fraction 0-1) var num = parseFloat(s); if (!isNaN(num) && num >= 0 && num < 1) { var totalMins = Math.round(num * 24 * 60); var hh = Math.floor(totalMins / 60); var mm = totalMins % 60; return String(hh).padStart(2,'0') + ':' + String(mm).padStart(2,'0'); } return s; } function doOptions(e) { return ContentService.createTextOutput('').setMimeType(ContentService.MimeType.TEXT); } // Standard JSON response (for POST) function buildResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // CORS-enabled response (for GET — cross-device sync) // Google Apps Script doGet responses are served via a redirect through // script.googleusercontent.com which strips custom headers, BUT // deploying as "Anyone" with "Execute as Me" makes GET requests work // cross-origin as long as we follow the redirect. We use no-cors on // POST and allow the browser to follow redirects on GET. function buildCorsResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ── Payroll sheet setup (unchanged) ────────────────────────────────────────── function setupPayrollSheet(ps) { ps.getRange(1,1,1,20).setValues([[ 'Employee Info','','Auto-Populated from Timesheet','','','','','','Manual Entry','', 'Leave (Auto-Populated)','','','','','','Manual Entry','','','' ]]); ps.getRange('A1:B1').merge(); ps.getRange('C1:J1').merge(); ps.getRange('K1:L1').merge(); ps.getRange('M1:P1').merge(); ps.getRange('Q1:T1').merge(); function grpCell(range, bg, fg) { var r = ps.getRange(range); r.setBackground(bg).setFontColor(fg).setFontFamily('Arial') .setFontSize(10).setFontWeight('bold') .setHorizontalAlignment('center').setVerticalAlignment('middle'); } grpCell('A1:B1','#1F4E3D','#FFFFFF'); grpCell('C1:J1','#1F4E3D','#FFFFFF'); grpCell('K1:L1','#C00000','#FFFFFF'); grpCell('M1:P1','#1F4E3D','#FFFFFF'); grpCell('Q1:T1','#C00000','#FFFFFF'); ps.setRowHeight(1,24); ps.getRange(2,1,1,20).setValues([[ 'Employee','Pay Period','Week 1 Hours','Week 2 Hours','2-Week Total','Auto Calc', 'Without OT','OT Hours','Mileage Units','Mileage $', 'Vacation Hours','Reason for Vacation','Holiday Hours','Reason for Holiday', 'Sick Hours','Reason for Sick','Notes','Bonus','Reimbursement','NOTES' ]]); ps.getRange('A2:H2').setBackground('#203864').setFontColor('#FFFFFF'); ps.getRange('K2:P2').setBackground('#203864').setFontColor('#FFFFFF'); ps.getRange('I2:J2').setBackground('#C00000').setFontColor('#FFFFFF'); ps.getRange('Q2:T2').setBackground('#C00000').setFontColor('#FFFFFF'); ps.getRange('A2:T2').setFontFamily('Arial').setFontSize(10).setFontWeight('bold') .setHorizontalAlignment('center').setVerticalAlignment('middle').setWrap(true); ps.setRowHeight(2,40); ps.getRange('A3:T3').merge(); ps.getRange('A3').setValue('● Auto-Calc = green when it matches 2-Week Total | ● OT hours highlighted | ● Leave hours in red | ● Yellow = manual entry columns'); ps.getRange('A3').setBackground('#E8EAF6').setFontColor('#444466').setFontSize(8) .setFontStyle('italic').setFontFamily('Arial'); ps.setRowHeight(3,16); var widths=[110,160,80,80,80,80,80,70,80,80,80,160,80,160,70,160,160,80,100,180]; for (var i=0;i<widths.length;i++) ps.setColumnWidth(i+1,widths[i]); var totalRow=29; ps.getRange(totalRow,1,1,2).merge(); ps.getRange(totalRow,1).setValue('TOTAL'); ps.getRange(totalRow,1,1,20).setBackground('#D9E1F2').setFontWeight('bold') .setFontFamily('Arial').setFontSize(10); var sumCols=[3,4,5,6,7,8,9,11,13,15,18,19]; for (var s=0;s<sumCols.length;s++) { var cl=columnLetter(sumCols[s]); ps.getRange(totalRow,sumCols[s]).setFormula('=SUM('+cl+'4:'+cl+(totalRow-1)+')'); } var remRow=totalRow+1; ps.getRange(remRow,1,1,2).merge(); ps.getRange(remRow,1).setValue('Remaining 26/26').setFontColor('#666688') .setFontStyle('italic').setFontSize(9).setFontFamily('Arial'); ps.getRange(remRow,1,1,20).setBackground('#F0F4FF'); ps.setFrozenRows(2); ps.setFrozenColumns(2); ps.setTabColor('#1F4E3D'); } function formatDataRow(ps, row) { var fullRow=ps.getRange(row,1,1,20); fullRow.setFontFamily('Arial').setFontSize(10).setVerticalAlignment('middle') .setBorder(null,null,true,null,null,null,'#C5CADF',SpreadsheetApp.BorderStyle.SOLID); ps.setRowHeight(row,18); ps.getRange(row,9,1,2).setBackground('#FFFFF0'); ps.getRange(row,17,1,4).setBackground('#FFFFF0'); ps.getRange(row,6).setFontColor('#00AA44').setFontWeight('bold'); ps.getRange(row,8).setFontColor('#E67E00').setFontWeight('bold'); ps.getRange(row,11).setFontColor('#CC0000'); ps.getRange(row,13).setFontColor('#CC0000'); ps.getRange(row,15).setFontColor('#CC0000'); var numCols=[3,4,5,6,7,8,9,11,13,15,18,19]; for (var n=0;n<numCols.length;n++) ps.getRange(row,numCols[n]).setHorizontalAlignment('center'); } function updateTotalRow(ps) { ps.getRange(29,1,1,20).setBackground('#D9E1F2').setFontWeight('bold'); } function columnLetter(col) { var letter=''; while (col>0) { var rem=(col-1)%26; letter=String.fromCharCode(65+rem)+letter; col=Math.floor((col-1)/26); } return letter; }
